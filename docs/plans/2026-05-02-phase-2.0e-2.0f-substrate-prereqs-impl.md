```python
def _check_no_downstream_executed(
    plan: Node,
    target_step_id: str,
    completed: set[str],
) -> None:
    """Phase 2.0e (#175): walk the subtree rooted at target_step_id and
    raise PlanEditDownstreamExecuted if any node id is in `completed`.

    The target itself counts — deleting a step whose effect has already
    committed is the canonical case the check guards.

    Args:
        plan: the Plan AST root (so we can locate the target subtree).
        target_step_id: the 32-hex Node.id of the step to be deleted.
        completed: tx.completed_step_ids — set of executed step ids.

    Raises:
        PlanEditDownstreamExecuted: if any id in the target subtree is
            in `completed`. Message names the first offending id (DFS).
    """
    from persistence.plan._errors import PlanEditDownstreamExecuted

    # Locate the target subtree via the existing pre-order DFS walk
    # (mirrors _splice_first / _walk_pre_order pattern used elsewhere).
    target_subtree = _find_subtree_pre_order(plan, target_step_id)
    if target_subtree is None:
        # delete_step itself raises StepIdNotFound for this case — let
        # that path own the error. Nothing to check.
        return

    for node in _walk(target_subtree):
        if node.id in completed:
            raise PlanEditDownstreamExecuted(
                f"delete_step: cannot remove step {target_step_id!r} — "
                f"its subtree contains node {node.id!r} which has already "
                f"executed in this transaction (tx.completed_step_ids). "
                f"Design § 4.1: 'only allowed if no downstream step has "
                f"executed.'"
            )
```
