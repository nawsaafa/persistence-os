"""Phase 2.2a — :fs/* effect handlers (read, write, glob, grep)."""
from __future__ import annotations
import base64
import hashlib
import re
from pathlib import Path
from typing import Any

from persistence.effect.runtime import Handler


class FsCapabilityDenied(Exception):
    """Raised when path resolves outside the configured project_root or scratch_dir."""


def _safe_resolve(p: str | Path, allowed_root: Path) -> Path:
    """Resolve `p` strict=False (path may not exist for writes); deny if real path
    is not relative to `allowed_root.resolve()`. Symlink-following is intentional."""
    resolved = Path(p).resolve(strict=False)
    if not resolved.is_relative_to(allowed_root.resolve()):
        raise FsCapabilityDenied(
            f"path {p!r} resolves to {resolved} which is outside project_root/scratch_dir"
        )
    return resolved


def _read_clause(project_root: Path):
    def _clause(args, k, ctx):
        path = _safe_resolve(args["path"], project_root)
        encoding = args.get("encoding", "utf-8")
        if encoding == "binary":
            data = path.read_bytes()
            return k({
                "bytes_or_text": base64.b64encode(data).decode("ascii"),
                "mtime": path.stat().st_mtime,
                "size": len(data),
                "sha256": hashlib.sha256(data).hexdigest(),
            })
        text = path.read_text(encoding=encoding)
        raw = text.encode(encoding)
        return k({
            "bytes_or_text": text,
            "mtime": path.stat().st_mtime,
            "size": len(raw),
            "sha256": hashlib.sha256(raw).hexdigest(),
        })
    return _clause


def _write_clause(scratch_dir: Path):
    def _clause(args, k, ctx):
        path = _safe_resolve(args["path"], scratch_dir)
        content = args["bytes_or_text"]
        mode = args.get("mode", "w")
        if mode in ("wb", "ab"):
            data = base64.b64decode(content)
            with path.open(mode) as f:
                bytes_written = f.write(data)
        else:
            data = content.encode("utf-8")
            with path.open(mode) as f:
                f.write(content)
                bytes_written = len(data)
        # sha256_after computed over the FULL file post-write
        full = path.read_bytes()
        return k({
            "bytes_written": bytes_written,
            "sha256_after": hashlib.sha256(full).hexdigest(),
        })
    return _clause


def _glob_clause(project_root: Path):
    def _clause(args, k, ctx):
        root = _safe_resolve(args["root"], project_root)
        pattern = args["pattern"]
        flags = args.get("flags", {})
        if flags.get("recursive", True):
            matches = [str(p) for p in root.rglob(pattern) if p.is_file()]
        else:
            matches = [str(p) for p in root.glob(pattern) if p.is_file()]
        return k({"matches": sorted(matches)})
    return _clause


def _grep_clause(project_root: Path):
    def _clause(args, k, ctx):
        root = _safe_resolve(args["root"], project_root)
        pattern = args["pattern"]
        flags = args.get("flags", {})
        re_flags = re.IGNORECASE if flags.get("ignore_case") else 0
        compiled = re.compile(re.escape(pattern) if flags.get("fixed_string") else pattern, re_flags)
        results: list[dict] = []
        for f in sorted(root.rglob("*")):
            if not f.is_file():
                continue
            try:
                for lineno, line in enumerate(f.read_text(errors="replace").splitlines(), 1):
                    if compiled.search(line):
                        results.append({"path": str(f), "line": lineno, "text": line})
            except (PermissionError, IsADirectoryError):
                continue
        results.sort(key=lambda r: (r["path"], r["line"]))
        return k({"matches": results})
    return _clause


def make_fs_handler(*, project_root: Path, scratch_dir: Path, name: str = "fs") -> Handler:
    return Handler(
        name=name,
        wraps={":fs/read", ":fs/write", ":fs/glob", ":fs/grep"},
        clauses={
            ":fs/read": _read_clause(project_root),
            ":fs/write": _write_clause(scratch_dir),
            ":fs/glob": _glob_clause(project_root),
            ":fs/grep": _grep_clause(project_root),
        },
    )
```

"""Phase 2.2a — :shell/exec effect handler with allowlist + sha256 versioning."""
from __future__ import annotations
import hashlib
import os
import subprocess
import time
from typing import Any

from persistence.effect.canonical import canonical_dumps
from persistence.effect.runtime import Handler


ALLOWLIST_V1: frozenset[str] = frozenset({
    "ls", "cat", "echo", "head", "tail", "wc",
    "grep", "find", "git",
    "python", "python3", "uv", "pip", "pytest",
})

ENV_DEFAULT: frozenset[str] = frozenset({
    "PATH", "HOME", "LANG", "LC_ALL", "LC_CTYPE",
})


def _allowlist_version(allowlist: frozenset[str]) -> str:
    return hashlib.sha256(
        canonical_dumps(sorted(allowlist)).encode()
    ).hexdigest()[:16]


ALLOWLIST_VERSION: str = _allowlist_version(ALLOWLIST_V1)


class ShellAllowlistDenied(Exception):
    """argv[0] is not in the allowlist."""


class ShellAllowlistVersionMismatch(Exception):
    """Recorded allowlist_version disagrees with handler's current version."""


def _shell_exec_clause(allowlist: frozenset[str], env_passthrough: frozenset[str]):
    current_version = _allowlist_version(allowlist)
    def _clause(args, k, ctx):
        argv = args["argv"]
        if not isinstance(argv, list) or not all(isinstance(s, str) for s in argv):
            raise TypeError(f"argv must be list[str], got {type(argv).__name__}")
        if not argv or argv[0] not in allowlist:
            raise ShellAllowlistDenied(
                f"argv[0]={argv[0]!r} not in allowlist (version {current_version})"
            )
        recorded_version = args["allowlist_version"]
        if recorded_version != current_version:
            raise ShellAllowlistVersionMismatch(
                f"recorded={recorded_version} current={current_version}"
            )
        cwd = args["cwd"]  # KeyError if missing — required
        env_subset = args.get("env_allowlist_subset", [])
        env = {k: os.environ[k] for k in env_subset if k in os.environ}
        timeout_s = args.get("timeout_s", 30.0)

        t0 = time.monotonic()
        try:
            cp = subprocess.run(
                argv, cwd=cwd, env=env,
                capture_output=True, text=True,
                timeout=timeout_s, shell=False,
            )
            return k({
                "exit": cp.returncode,
                "stdout": cp.stdout,
                "stderr": cp.stderr,
                "wall_clock_ms": int((time.monotonic() - t0) * 1000),
            })
        except subprocess.TimeoutExpired as e:
            return k({
                "exit": -9,
                "stdout": (e.stdout or b"").decode("utf-8", errors="replace") if isinstance(e.stdout, bytes) else (e.stdout or ""),
                "stderr": (e.stderr or b"").decode("utf-8", errors="replace") if isinstance(e.stderr, bytes) else (e.stderr or ""),
                "wall_clock_ms": int((time.monotonic() - t0) * 1000),
            })
    return _clause


def make_shell_handler(
    *,
    allowlist: frozenset[str] = ALLOWLIST_V1,
    env_passthrough: frozenset[str] = ENV_DEFAULT,
    name: str = "shell",
) -> Handler:
    return Handler(
        name=name,
        wraps={":shell/exec"},
        clauses={":shell/exec": _shell_exec_clause(allowlist, env_passthrough)},
    )

    t0 = dt.datetime.now(dt.timezone.utc)  # noqa: wall-clock

    def _record(result: Any | None, error: str | None) -> None:
        latency_ms = int((dt.datetime.now(dt.timezone.utc) - t0).total_seconds() * 1000)
        self.substrate.fact.transact([{
            "e": uuid.uuid4().hex,
            "a": ":act/result",
            "v": canonical_dumps({
                "op": op,
                "args_hash": canonical_hash(args),
                "result_summary": _summarize_result(result) if result is not None else None,
                "error": error,
                "latency_ms": latency_ms,
            }),
            "valid_from": t0,
        }])

    try:
        result = self.substrate.effect.perform(op, args)
    except Exception as e:
        _record(None, f"{type(e).__name__}: {e}")
        raise

    _record(result, None)


def _summarize_result(r: Any) -> Any:
    """Truncate large blobs; keep small results intact for prompt re-read."""
    if isinstance(r, dict):
        out = {}
        for k, v in r.items():
            if isinstance(v, str) and len(v) > 512:
                out[k] = v[:256] + "...[truncated]..." + v[-256:]
            else:
                out[k] = v
        return out
    return r
```
