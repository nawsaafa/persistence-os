```python
def _code_run_clause(args, _k, _ctx):
    """Phase 2.2b T2 — substantive-return clause for :code/run.

    Reuses run_subprocess (extracted in T1). Returns full result dict
    including output_hash; audit middleware computes args_hash over the
    LLM-emitted op args (source-shape) and result_hash over the full
    result dict (which includes wall_clock_ms — non-deterministic by design).
    """
    source = args["source"]
    stdin = args.get("stdin", "")
    timeout_seconds = args.get("timeout_seconds", 5.0)
    memory_mb = args.get("memory_mb", 128)
    env = args.get("env")

    stdout, stderr, exit_code, wall_clock_ms = run_subprocess(
        source=source, stdin=stdin,
        timeout_seconds=timeout_seconds, memory_mb=memory_mb, env=env,
    )
    return {
        "stdout": stdout, "stderr": stderr,
        "exit_code": exit_code, "wall_clock_ms": wall_clock_ms,
        "output_hash": output_hash(stdout, stderr, exit_code),
    }


def make_code_run_dispatch_handler(*, name="code-run"):
    """Phase 2.2b T2 — bottom-of-stack handler for :code/run.

    Forced spec deviation #1: factory takes (name) only, NOT (project_root,
    scratch_dir) as design § 3 suggested. Neither param is referenced —
    YAGNI-correct.
    """
    return Handler(
        name=name,
        wraps={":code/run"},
        clauses={":code/run": _code_run_clause},
    )
```

```python
from pathlib import Path
from typing import Any
from persistence.effect.handlers.fs import _safe_resolve
from persistence.effect.handlers.shell import ALLOWLIST_VERSION
from persistence.effect.runtime import Handler, mask, perform

_ENV_SUBSET = ["PATH", "HOME", "LANG", "LC_ALL", "LC_CTYPE"]
_GIT_PREFIX = ["git", "-c", "color.ui=false", "-c", "core.pager=cat"]
_ALLOWED_LOG_FORMATS = frozenset({"oneline", "short", "medium"})
_DEFAULT_TIMEOUT_S = 30.0


class GitArgValidation(ValueError):
    """Raised when :git/* args fail validation (NOT a git error string)."""


def _resolve_cwd(args, project_root):
    cwd_raw = args.get("cwd", str(project_root))
    return _safe_resolve(cwd_raw, project_root)


def _diff_clause_factory(project_root, audit_handler_name):
    def clause(args, _k, _ctx):
        cwd = _resolve_cwd(args, project_root)
        ref = args.get("ref", "HEAD")
        cached = args.get("cached", False)
        paths = sorted(args.get("paths", []))
        argv = list(_GIT_PREFIX) + ["diff", "--no-color"]
        if cached:
            argv.append("--cached")
        argv.append(ref)
        argv.append("--")
        argv.extend(paths)
        with mask(audit_handler_name):
            return perform(":shell/exec",
                           argv=argv, cwd=str(cwd),
                           allowlist_version=ALLOWLIST_VERSION,
                           env_allowlist_subset=_ENV_SUBSET,
                           timeout_s=_DEFAULT_TIMEOUT_S)
    return clause

# Similar factories for _status_clause_factory, _log_clause_factory, _commit_clause_factory.
# _log validates n in [1, 1000] and format in _ALLOWED_LOG_FORMATS.
# _commit validates non-empty message + non-empty paths + each path resolves
# inside project_root.

def make_git_handler(*, project_root, audit_handler_name="audit"):
    return Handler(
        name="git",
        wraps={":git/diff", ":git/status", ":git/log", ":git/commit"},
        clauses={
            ":git/diff": _diff_clause_factory(project_root, audit_handler_name),
            ":git/status": _status_clause_factory(project_root, audit_handler_name),
            ":git/log": _log_clause_factory(project_root, audit_handler_name),
            ":git/commit": _commit_clause_factory(project_root, audit_handler_name),
        },
    )

__all__ = ["GitArgValidation", "make_git_handler"]
```

```python
def _render_latest_action(action):
    """LD3 — render the most-recent action with explicit field formatting.

    Stable shape: op, error, result_summary fields rendered explicitly
    (NOT json.dumps[:200] which truncated before reaching result_summary).
    String result fields keep _summarize_result's ~600-char truncation;
    list fields render as 'count=N, first_3=[...]'; non-string non-list
    fields render via repr() truncated at 200 chars.
    """
    parts = ["Latest action output (iter "
             + str(action.get("iter_count", "?")) + "):"]
    parts.append(f"  op: {action.get('op', '?')}")
    parts.append(f"  error: {action.get('error') or 'none'}")
    rs = action.get("result_summary")
    if rs is None:
        parts.append("  (no result_summary — exception path)")
    elif isinstance(rs, dict):
        for k, v in rs.items():
            if isinstance(v, str):
                parts.append(f"  {k}: |\n    {v}")
            elif isinstance(v, list):
                parts.append(f"  {k}: count={len(v)}, first_3={v[:3]!r}")
            else:
                parts.append(f"  {k}: {v!r}"[:202])
    else:
        parts.append(f"  result_summary: {rs!r}"[:202])
    return parts
```
